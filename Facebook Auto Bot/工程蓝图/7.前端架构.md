# 7. 前端架构

## 架构设计原则

### 核心原则
1. **PWA优先**：支持离线功能，可安装到主屏幕
2. **移动优先**：优先考虑移动端体验
3. **性能优先**：快速加载，流畅交互
4. **可访问性**：支持屏幕阅读器和键盘导航
5. **渐进增强**：基础功能在所有设备上可用

### 技术原则
1. **组件化**：复用组件，减少重复代码
2. **状态集中**：统一的状态管理
3. **类型安全**：全面使用TypeScript
4. **自动化测试**：单元测试和E2E测试
5. **代码质量**：严格的代码规范和检查

```typescript
interface UserStore {

const useUserStore = create<UserStore>((set, get) => ({
  user: null,
  isAuthenticated: false,
  isLoading: false,
```

### API服务层
```typescript
// API客户端
const apiClient = axios.create({
  baseURL: import.meta.env.VITE_API_URL,
  timeout: 30000,
  headers: {
    'Content-Type': 'application/json',
  },
});

// 请求拦截器（添加Token）
apiClient.interceptors.request.use((config) => {
  const token = localStorage.getItem('access_token');
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;
});

// 响应拦截器（处理错误）
apiClient.interceptors.response.use(
  (response) => response,
  (error) => {
    if (error.response?.status === 401) {
      // Token过期，跳转到登录页
      localStorage.removeItem('access_token');
      window.location.href = '/auth/login';
    }
    return Promise.reject(error);
  }
);

// 账号API服务
export const accountApi = {
  // 获取账号列表
  getAccounts: (params?: GetAccountsParams) =>
    apiClient.get<ApiResponse<FacebookAccount[]>>('/accounts', { params }),
  
  // 创建账号
  createAccount: (data: CreateAccountDto) =>
    apiClient.post<ApiResponse<FacebookAccount>>('/accounts', data),
  
  // 更新账号
  updateAccount: (id: string, data: UpdateAccountDto) =>
    apiClient.put<ApiResponse<FacebookAccount>>(`/accounts/${id}`, data),
  
  // 删除账号
  deleteAccount: (id: string) =>
    apiClient.delete<ApiResponse<void>>(`/accounts/${id}`),
};
```

### 离线功能
```typescript
// 离线检测Hook
const useOffline = () => {
  const [isOffline, setIsOffline] = useState(!navigator.onLine);
  
  useEffect(() => {
    const handleOnline = () => setIsOffline(false);
    const handleOffline = () => setIsOffline(true);
    
    window.addEventListener('online', handleOnline);
    window.addEventListener('offline', handleOffline);
    
    return () => {
      window.removeEventListener('online', handleOnline);
      window.removeEventListener('offline', handleOffline);
    };
  }, []);
  
  return isOffline;
};

// 离线队列
class OfflineQueue {
  private queue: OfflineOperation[] = [];
  
  addOperation(operation: OfflineOperation) {
    this.queue.push(operation);
    this.saveToStorage();
  }
  
  async processQueue() {
    while (this.queue.length > 0) {
      const operation = this.queue[0];
      try {
        await this.executeOperation(operation);
        this.queue.shift();
      } catch (error) {
        console.error('操作执行失败:', error);
        break;
      }
    }
    this.saveToStorage();
  }
  
  private saveToStorage() {
    localStorage.setItem('offline_queue', JSON.stringify(this.queue));
  }
}
```
